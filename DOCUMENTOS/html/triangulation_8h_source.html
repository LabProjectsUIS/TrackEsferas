39  *
40  * Function that computes the position of a set on N points given the left and right image projections.
41  * The cameras are assumed to be calibrated, intrinsically, and extrinsically.
42  *
43  * Input:
44  * xL : 2xN matrix of pixel coordinates in the left image
45  * xR : 2xN matrix of pixel coordinates in the right image
46  * om, T : rotation vector and translation vector between right and left cameras(output of stereo calibration)
47  * fc_left, cc_left, ... : intrinsic parameters of the left camera(output of stereo calibration)
48  * fc_right, cc_right, ... : intrinsic parameters of the right camera(output of stereo calibration)
49  *
50  * Output :
51  *
52  * XL : 3xN matrix of coordinates of the points in the left camera reference frame
53  * XR : 3xN matrix of coordinates of the points in the right camera reference frame
54  *
55  * Note : XR and XL are related to each other through the rigid motion equation : XR = R * XL + T, where R = rodrigues(om)
56  * For more information, visit http ://www.vision.caltech.edu/bouguetj/calib_doc/htmls/example5.html
57  *
58  *
59  * (c)Jean - Yves Bouguet - Intel Corporation - April 9th, 2003*/

67 
68  //-- - Rotation matrix corresponding to the rigid motion between left and right cameras :
69  Rodrigues(om, R);
70

72  u = R * xt;

77  dotProduct(u, xtt, u2);

79  DD = n_xt2.mul(n_xtt2) - u2;
80

84

86  NN2 = n_xt2.mul(dot_xttT) - dot_uT.mul(dot_xttu);
87

89  divide(NN2.t(), DD.t(), Ztt);
