6 This code defines a set of functions to align rigid objects.
7 Both objects, from now on referred as source and target,
8 are Point Clouds.
9 
10 Registration is performed in two steps. First, the
11 source is brought near the target in a process involving
12 voxelization and a feature-based RANSAC; prior to
13 features estimation, normals are to be computed. Second,
14 an ICP algorithm is applied to source in order to match
15 it to target.

89 void bring_cloud_near(PointCloudT::Ptr object, PointCloudT::Ptr scene, Eigen::Matrix4f& transformation, int flag) {

94  RadiusOutlierRemovalT outlierRemoval;

96  VoxelGridT grid;

98 
99  const float leaf = 10;
100  double radius = 10;
101 
102  switch (flag) {
103  case 1:
104  // Remove outliers
105  outlierRemoval.setInputCloud(object);
106  // Every point must have 10 neigbors within 10mm, or it will be removed.
107  outlierRemoval.setRadiusSearch(10);
108  outlierRemoval.setMinNeighborsInRadius(10);
109  outlierRemoval.filter(*object);
110  radius = 5;
111  nest.setSearchMethod(kdtree);
112  break;
113  case 2:
114  // Downsample
115  grid.setLeafSize(leaf, leaf, leaf);
116  grid.setInputCloud(object);
117  grid.filter(*object);
118  grid.setInputCloud(scene);
119  grid.filter(*scene);
120  radius = 0.1;
121  break;
122  default:
123  break;
124  }
125 
126  // Estimate normals for scene
127  nest.setRadiusSearch(radius);
128  nest.setInputCloud(scene);
129  nest.compute(*scene);
130 
131  // Estimate features
132  FeatureEstimationT fest;
133  fest.setRadiusSearch(50);
134  fest.setInputCloud(object);
135  fest.setInputNormals(object);
136  fest.compute(*object_features);
137  fest.setInputCloud(scene);
138  fest.setInputNormals(scene);
139  fest.compute(*scene_features);
140 
141  // Perform alignment
142  SampleConsensusPrerejectiveT align;
143  align.setInputSource(object);
144  align.setSourceFeatures(object_features);
145  align.setInputTarget(scene);
146  align.setTargetFeatures(scene_features);
147  align.setMaximumIterations(50000); // Number of RANSAC iterations
148  align.setNumberOfSamples(10); // Number of points to sample for generating/prerejecting a pose
149  align.setCorrespondenceRandomness(10); // Number of nearest features to use
150  align.setSimilarityThreshold(0.0001f); // Polygonal edge length similarity threshold
151  align.setMaxCorrespondenceDistance(1.01f * leaf); // Inlier threshold
152  align.setInlierFraction(0.95f); // Required inlier fraction for accepting a pose hypothesis
153  {
154  pcl::ScopeTime t("Alignment");
155  align.align(*object_aligned);
156  }
157 
158 
159  if (align.hasConverged())
160  {
161  pcl::transformPointCloud(*object, *object, align.getFinalTransformation());
162  transformation = align.getFinalTransformation().inverse();
163  }
164  else
165  {
166  pcl::console::print_error("Alignment failed!\n");
167  }
168 }
