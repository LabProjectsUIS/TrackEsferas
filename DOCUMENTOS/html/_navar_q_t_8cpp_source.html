646 bool NavarQT::readParams(std::string fileName) {
647  struct stat buffer;
648  if (stat(fileName.c_str(), &buffer) == 0) {
649  cv::FileStorage fs(fileName, cv::FileStorage::READ);
650  cv::Mat_<double> err(1, 1);
651  //double err;
