172 void GUIUpdater::getAngles(cv::Mat_<double>& vector, cv::Mat_<double>& angles) {
173  double x, y, z, ax, ay, az;
174  x = vector(0, 0);
175  y = vector(1, 0);
176  z = vector(2, 0);
177 
178  ax = atan2(sqrt(y*y + z*z), x);
179  ax = ax * 180 / M_PI;
180  ay = atan2(sqrt(x*x + z*z), y);
181  ay = ay * 180 / M_PI;
182  az = atan2(sqrt(y*y + x*x), z);
183  az = az * 180 / M_PI;
184 
185  angles = (cv::Mat_<double>(3, 1) << ax, ay, az);
186 }
